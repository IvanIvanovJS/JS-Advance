import { expect } from "chai";
import { deckOfCards } from "./deckOfCards.js"
 
describe("Checking cards", () => {
    it("Is invalid card input", () => {
        expect(deckOfCards(['5S', '3D', 'QD', '1C'])).to.be.equal("Invalid card: 1C")
    });
    it("is propper input working", () => {
        expect(deckOfCards(['AS', '10D', 'KH', '2C'])).to.be.equal("A♠ 10♦ K♥ 2♣")
    })
    it("should return empty string", () => {
        expect(deckOfCards([])).to.be.equal("")
    })
    it("should return message for invalid card", () => {
        expect(deckOfCards(["ZZ"])).to.be.equal("Invalid card: ZZ")
    })
    it("should return message for invalid card with string", () => {
        expect(() => deckOfCards("123")).to.throw("cards is not iterable")
    })
    it("should return message for invalid card with number", () => {
        expect(() => deckOfCards(1)).to.throw("cards is not iterable")
    })
    it("should return message for card.slice is not a function", () => {
        expect(() => deckOfCards([123])).to.throw("card.slice is not a function")
    })
    it("should return message for invalid card", () => {
        expect(deckOfCards(["aisdkaskdask", "qwofaksfkask", 123])).to.be.equal("Invalid card: aisdkaskdask")
    })
 
 
})
